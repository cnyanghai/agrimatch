    <script>
        function updateImpact() {
            const price = document.getElementById('counter-price').value;
            const impactProfit = document.getElementById('impact-profit');
            
            // 简单的联动逻辑模拟
            const diff = (price - 860) * 50000;
            if (diff >= 0) {
                impactProfit.innerText = "+ ¥" + diff.toLocaleString();
                impactProfit.parentElement.parentElement.className = "p-4 bg-emerald-50 rounded-xl border border-emerald-100";
            } else {
                impactProfit.innerText = "- ¥" + Math.abs(diff).toLocaleString();
                impactProfit.parentElement.parentElement.className = "p-4 bg-red-50 rounded-xl border border-red-100";
            }
        }

        function submitCounter() {
            const overlay = document.getElementById('success-overlay');
            overlay.classList.remove('opacity-0', 'pointer-events-none');
            setTimeout(() => {
                location.reload();
            }, 2000);
        }

        function cancelCounter() {
            if(confirm("确定放弃当前的修改吗？")) {
                location.reload();
            }
        }
    </script>
