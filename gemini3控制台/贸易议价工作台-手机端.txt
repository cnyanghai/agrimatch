    <script>
        let deliveryMode = '港口自提';

        function switchTab(tab) {
            const chatView = document.getElementById('view-chat');
            const negView = document.getElementById('view-neg');
            const footer = document.getElementById('footer-input');
            const tabChat = document.getElementById('tab-chat');
            const tabNeg = document.getElementById('tab-neg');

            if (tab === 'chat') {
                chatView.classList.remove('hidden');
                negView.classList.add('hidden');
                footer.classList.remove('hidden');
                tabChat.className = "flex-1 py-3 text-sm font-medium border-b-2 border-blue-600 text-blue-600";
                tabNeg.className = "flex-1 py-3 text-sm font-medium border-b-2 border-transparent text-gray-500";
            } else {
                chatView.classList.add('hidden');
                negView.classList.remove('hidden');
                footer.classList.add('hidden');
                tabChat.className = "flex-1 py-3 text-sm font-medium border-b-2 border-transparent text-gray-500";
                tabNeg.className = "flex-1 py-3 text-sm font-medium border-b-2 border-blue-600 text-blue-600";
            }
        }

        function setDelivery(mode) {
            deliveryMode = mode;
            const b1 = document.getElementById('del-1');
            const b2 = document.getElementById('del-2');
            if(mode === '港口自提') {
                b1.className = "flex-1 py-2 text-xs rounded-lg border border-blue-600 bg-blue-50 text-blue-600 font-bold";
                b2.className = "flex-1 py-2 text-xs rounded-lg border border-gray-200 bg-white text-gray-500";
            } else {
                b2.className = "flex-1 py-2 text-xs rounded-lg border border-blue-600 bg-blue-50 text-blue-600 font-bold";
                b1.className = "flex-1 py-2 text-xs rounded-lg border border-gray-200 bg-white text-gray-500";
            }
        }

        function sendMessage() {
            const input = document.getElementById('chat-input');
            if(!input.value.trim()) return;
            
            const msgHtml = `
                <div class="flex justify-end message-in">
                    <div class="max-w-[85%] px-4 py-3 bg-blue-600 text-white rounded-2xl rounded-tr-none text-sm shadow-md">
                        ${input.value}
                        <div class="text-[9px] mt-1 opacity-60 text-right">${new Date().toLocaleTimeString([], {hour:'2-digit', minute:'2-digit'})}</div>
                    </div>
                </div>
            `;
            document.getElementById('dynamic-messages').insertAdjacentHTML('beforeend', msgHtml);
            input.value = '';
            scrollToBottom();
        }

        function submitNegotiation() {
            const price = document.getElementById('form-price').value;
            const qty = document.getElementById('form-qty').value;
            
            const cardHtml = `
                <div class="flex justify-end message-in">
                    <div class="bg-orange-50 border border-orange-200 rounded-2xl p-4 w-full max-w-[260px] shadow-sm">
                        <div class="flex items-center gap-2 mb-3 text-orange-700 font-bold text-xs">
                            <svg width="14" height="14" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><rect x="2" y="5" width="20" height="14" rx="2"></rect><line x1="2" y1="10" x2="22" y2="10"></line></svg>
                            发起了一项议价方案
                        </div>
                        <div class="space-y-2 text-xs border-b border-orange-100 pb-3">
                            <div class="flex justify-between">
                                <span class="text-gray-500">期望价格:</span>
                                <span class="font-bold text-orange-600 text-sm">¥${price} / 吨</span>
                            </div>
                            <div class="flex justify-between">
                                <span class="text-gray-500">拟购数量:</span>
                                <span class="font-medium">${qty} 吨</span>
                            </div>
                            <div class="flex justify-between">
                                <span class="text-gray-500">物流方式:</span>
                                <span class="font-medium">${deliveryMode}</span>
                            </div>
                        </div>
                        <div class="mt-3 text-[10px] text-orange-400 text-center italic">等待对方确认或二次议价</div>
                    </div>
                </div>
            `;
            
            switchTab('chat');
            document.getElementById('dynamic-messages').insertAdjacentHTML('beforeend', cardHtml);
            scrollToBottom();
        }

        function scrollToBottom() {
            const area = document.getElementById('content-area');
            area.scrollTop = area.scrollHeight;
        }

        // 处理回车发送
        document.getElementById('chat-input').addEventListener('keypress', function (e) {
            if (e.key === 'Enter') sendMessage();
        });
    </script>
